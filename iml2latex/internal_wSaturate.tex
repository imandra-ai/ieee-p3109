 X \\}\\
\Case{\wSaturate(\SatFinite, \any, +\infty, \any, \any) \gives  \Mhi}\\
\Case{\wSaturate(\SatFinite, \any, -\infty, \any, \any) \gives  \Mlo}\\
\Case{\wSaturate(\SatFinite, \any, X, \any, \any) ~\If X < \Mlo\gives  \Mlo}\\
\Case{\wSaturate(\SatFinite, \any, X, \any, \any) ~\If X > \Mhi\gives  \Mhi \\}\\
\Case{\wSaturate(\SatPropagate, \any, +\infty, \any, \Extended) \gives  +\infty}\\
\Case{\wSaturate(\SatPropagate, \any, +\infty, \any, \any) \gives  \Mhi}\\
\Case{\wSaturate(\SatPropagate, \any, -\infty, \Signed, \Extended) \gives
 -\infty}\\
\Case{\wSaturate(\SatPropagate, \any, -\infty, \any, \any) \gives  \Mlo}\\
\Case{\wSaturate(\SatPropagate, \any, X, \any, \any) ~\If X < \Mlo\gives  \Mlo}\\
\Case{\wSaturate(\SatPropagate, \any, X, \any, \any) ~\If X > \Mhi\gives  \Mhi \\}\\
\Case{\wSaturate(\SatNone, \TowardZero \Or \TowardNegative, X, \any, \any) ~\If
X > \Mhi \And X \neq +\infty\gives  \Mhi}\\
\Case{\wSaturate(\SatNone, \TowardZero \Or \TowardPositive, X, \any, \any) ~\If
X < \Mlo \And X \neq -\infty\gives  \Mlo}\\

\Case{\wSaturate(\SatNone, \any, -\infty, \Signed, \Extended) \gives  -\infty}\\
\Case{\wSaturate(\SatNone, \any, -\infty, \Unsigned, \Extended) \gives  \NaN}\\

\Mhi}\\
\Case{\wSaturate(\SatNone, \any, X, \Signed, \Extended) ~\If X < \Mlo\gives  -\infty}\\
\Case{\wSaturate(\SatNone, \any, X, \Unsigned, \Extended) ~\If X < \Mlo\gives  \NaN}\\
